! ============================================================
! SWANS - Cisco c7200 CE Router (VRF SWANS, vlan1804)
! Loopback0: 172.16.21.2/32 (advertised in global IS-IS)
! Loopback100: 192.168.100.1/24 (customer local, reachable over VRF)
! Uplink: fa0/0 trunk to PE4 fa4/1
!   .10  global IS-IS+MPLS link    -- 10.0.21.4/30
!   .20  customer (PE4 VRF SWANS)  -- 10.0.22.4/30
! Sites: fa1/0 -> SWANS_SITE_A (10.101.0.0/30, lan 192.168.103.0/24)
!        fa2/0 -> SWANS_SITE_B (10.101.0.4/30, lan 192.168.104.0/24)
! ============================================================

hostname SWANS
no ip domain-lookup
ip cef
mpls label protocol ldp
!
enable secret 0 lab
!
cdp run
!
! -- Syslog source + NetFlow v9 (LAB-2 telemetry)
logging source-interface Loopback0
ip flow-export version 9
ip flow-export source Loopback0
ip flow-export destination 10.2.0.114 9996
ip flow-cache timeout active 1
!
interface Loopback0
 description ROUTER-ID
 ip address 172.16.21.2 255.255.255.255
 ip flow ingress
 ip router isis LAB
!
interface Loopback100
 description CUSTOMER-LOCAL
 ip address 192.168.100.1 255.255.255.0
 ip flow ingress
!
interface FastEthernet0/0
 description TRUNK -- PE4 fa4/1 (vlan10=global, vlan20=VRF SWANS)
 no ip address
 no shutdown
!
interface FastEthernet0/0.10
 description CORE -- PE4 fa4/1.10 (global IS-IS+MPLS)
 encapsulation dot1Q 10
 ip address 10.0.21.6 255.255.255.252
 ip flow ingress
 ip router isis LAB
 isis network point-to-point
 mpls ip
!
interface FastEthernet0/0.20
 description CE -- PE4 fa4/1.20 (VRF SWANS customer side)
 encapsulation dot1Q 20
 ip address 10.0.22.6 255.255.255.252
 ip flow ingress
!
interface FastEthernet1/0
 description CE -- SWANS_SITE_A ether1
 ip address 10.101.0.1 255.255.255.252
 ip flow ingress
 no shutdown
!
interface FastEthernet2/0
 description CE -- SWANS_SITE_B ether2
 ip address 10.101.0.5 255.255.255.252
 ip flow ingress
 no shutdown
!
ip route 0.0.0.0 0.0.0.0 10.0.22.5
!
! Static routes to the site LANs reachable through each branch router.
ip route 192.168.103.0 255.255.255.0 10.101.0.2
ip route 192.168.104.0 255.255.255.0 10.101.0.6
!
router isis LAB
 net 49.0001.0000.AC10.1502.00
 is-type level-2-only
 metric-style wide
 passive-interface Loopback0
!
mpls ldp router-id Loopback0 force
!
snmp-server community public RO
snmp-server location EVE-NG Lab
!
logging host 10.2.0.114
logging trap informational
logging facility local6
service timestamps log datetime msec localtime
!
line con 0
 privilege level 15
 logging synchronous
line vty 0 4
 login
 password lab
!
end
